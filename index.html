const express = require("express"); const fetch = require("node-fetch"); const cors = require("cors"); const app = express(); app.use(cors()); app.use(express.json()); let enquiries = []; app.post("/enquiry", (req, res) => { enquiries.push({...req.body, status: "Pending"}); res.send("Saved"); }); app.get("/enquiries", (req, res) => { res.json(enquiries); }); app.post("/approve", async (req, res) => { const { index } = req.body; enquiries[index].status = "Approved"; await sendSMS(enquiries[index].phone, "Your request has been APPROVED - Sunny CSC Centre"); res.send("Approved"); }); app.post("/reject", async (req, res) => { const { index } = req.body; enquiries[index].status = "Rejected"; await sendSMS(enquiries[index].phone, "Your request has been REJECTED - Sunny CSC Centre"); res.send("Rejected"); }); async function sendSMS(phone, message){ await fetch("https://www.fast2sms.com/dev/bulkV2", { method: "POST", headers: { authorization: "YOUR_API_KEY", "Content-Type": "application/json" }, body: JSON.stringify({ route: "q", message, language: "english", numbers: phone }) }); } app.listen(3000, () => console.log("Server running"));
